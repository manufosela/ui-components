Simulated Loading

Non-linear progress with status change on completion

Show code
<circle-percent id="loader" percent="0" stroke-color="#8b5cf6" title="Loading..."> </circle-percent> <script type="module"> const el = document.getElementById('loader'); const steps = [ [12,300],[25,500],[30,200],[45,800], [48,1200],[60,400],[72,300],[78,600], [85,900],[90,200],[95,400],[100,300], ]; const wait = ms => new Promise(r => setTimeout(r, ms)); async function simulateLoad() { for (const [target, delay] of steps) { await wait(delay); const s = el.percent; for (let i=1; i<=target-s; i++) { await wait(30); el.percent = s+i; } } el.title = 'Load OK ✓'; el.strokeColor = '#10b981'; } simulateLoad(); </script>

Animated Progress

Simulated loading with non-linear progress and status text

const el = document.querySelector('circle-percent'); el.title = 'Loading...'; el.strokeColor = '#22c55e'; // Non-linear steps: [targetPercent, delayMs] const steps = [ [12, 300], [25, 500], [30, 200], [45, 800], [48, 1200], // pause — simulates a slow resource [60, 400], [72, 300], [78, 600], [85, 900], // another pause [90, 200], [95, 400], [100, 300], ]; async function simulateLoad() { for (const [target, delay] of steps) { await new Promise(r => setTimeout(r, delay)); // Animate smoothly from current to target const start = el.percent; const diff = target - start; for (let i = 1; i <= diff; i++) { await new Promise(r => setTimeout(r, 30)); el.percent = start + i; } } el.title = 'Load OK'; el.strokeColor = '#10b981'; } simulateLoad();
